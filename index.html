        [bits 32]

a1       equ 5           ; Pierwszy wyraz ciągu (np. 5)

         mov ebp, ebx    ; Zapisanie wskaźnika API

         ; --- OBLICZENIA ---
         mov eax, n      ; eax = n
         dec eax         ; eax = n - 1

         ; --- WYPISANIE ---

         call getaddr

         inc esi             ; Zwiększamy indeks
         loop .loop          ; Zmniejsz ecx, skacz jeśli ecx > 0

         ; --- WYPISANIE ---
         push eax
         
         call getaddr

         db "Iloczyn elementow = %d", 0xA, 0

Zadanie 4 – r_wyraz.asm

Napisz program r_wyraz.asm obliczający rekurencyjnie n-ty wyraz ciągu arytmetycznego dla zadanych stałych całkowitych a1, r, n. wyraz(1) = a(1) wyraz(n) = wyraz(n-1) + r


        [bits 32]

a1       equ 5           ; Pierwszy wyraz
r        equ 3           ; Różnica
n        equ 4           ; Szukany n-ty wyraz

         jmp start       ; Omijamy kod funkcji

; ==========================================
; FUNKCJA REKURENCYJNA
; Wejście: ECX = n
; Wyjście: EAX = wynik
; ==========================================
oblicz_rek:
         cmp ecx, 1      ; Sprawdzenie warunku stopu (n <= 1)
         jle bazowy
         
         push ecx        ; Zapisanie obecnego 'n' na stos
         dec ecx         ; Zmniejszenie 'n' (n-1)
         call oblicz_rek ; Wywołanie rekurencyjne dla n-1
         pop ecx         ; Odtworzenie 'n'
         
         ; Po powrocie z rekurencji w EAX mamy wartość wyraz(n-1)
         add eax, r      ; eax = wyraz(n-1) + r
         ret

bazowy:
         mov eax, a1     ; Dla n=1 zwracamy po prostu a1
         ret

; ==========================================
; GŁÓWNY PROGRAM
; ==========================================
start:
         mov ebp, ebx    ; Wskaźnik API
         
         mov ecx, n      ; Przekazanie 'n' do funkcji
         call oblicz_rek ; Start machiny
         
         push eax        ; Wynik -> stos
         
         call getaddr
format:
         db "Wyraz (rekurencja) = %d", 0xA, 0
getaddr:

         call [ebp+3*4]  ; printf
         add esp, 4
         
         push 0
         call [ebp+0*4]  ; exit(0)

Zadanie 5 – funkcja.asm

Napisz program funkcja.asm obliczający wartość funkcji f(x) = x - 3*sin(x) dla argumentu zmienno-pozycyjnego x typu double.


        [bits 32]

x        dq 1.5           ; Przykładowy x typu double (dq = define quadword = 8 bajtów)
trzy     dd 3             ; Stała całkowita 3

         mov ebp, ebx     ; Wskaźnik API

         ; --- OBLICZENIA FPU ---
         fld qword [x]    ; Wrzucamy 'x' na stos FPU (st0 = x)
         fld st0          ; Kopiujemy szczyt stosu (st0 = x, st1 = x)
         
         fsin             ; Liczymy sinus z st0 (st0 = sin(x), st1 = x)
         
         fild dword [trzy]; Wrzucamy całkowite 3 (st0 = 3, st1 = sin(x), st2 = x)
         fmulp st1, st0   ; Mnożymy i zdejmujemy: st0 = 3 * sin(x), st1 = x
         
         fsubp st1, st0   ; Odejmujemy st1 - st0 i zdejmujemy: st0 = x - 3*sin(x)

         ; --- WYPISYWANIE DOUBLE ---
         ; Uwaga: funkcja printf z "%f" oczekuje, że double zajmie 8 bajtów na stosie!
         sub esp, 8       ; Robimy 8 bajtów wolnego miejsca na zwykłym stosie procesora
         fstp qword [esp] ; Zdejmujemy wynik z koprocesora i zapisujemy go w tym wolnym miejscu

         call getaddr
format:
         db "f(x) = %f", 0xA, 0
getaddr:

         call [ebp+3*4]   ; printf(format, x_double)
         add esp, 12      ; Czyszczenie: 8 bajtów za double + 4 bajty za wskaźnik 'format'

         push 0
         call [ebp+0*4]   ; exit(0)

        [bits 32]

a1       equ 2

         mov ecx, a1

         imul ecx        ; eax = a2 * q

Napisz program funkcja.asm wyliczający wartość funkcji f(x) = 5*cos(x) - x dla argumentu zmienno-pozycyjnego x typu double.


        [bits 32]

x        dq 0.0          ; Wpiszmy 0.0, żeby łatwo sprawdzić. 5*cos(0) - 0 = 5*1 - 0 = 5.0
piec     dd 5            ; Stała całkowita do mnożenia

         mov ebp, ebx

         ; --- OBLICZENIA FPU ---
         fld qword [x]    ; Wrzucamy x (st0 = x)
         fld st0          ; Kopiujemy szczyt (st0 = x, st1 = x)
         
         fcos             ; Liczymy kosinus (st0 = cos(x), st1 = x)
         
         fild dword [piec]; (st0 = 5, st1 = cos(x), st2 = x)
         fmulp st1, st0   ; Mnożymy i zdejmujemy (st0 = 5*cos(x), st1 = x)
         
         ; Mamy st0 = 5*cos(x) oraz st1 = x. 
         ; Chcemy wynik (st0 - st1) wrzucić do st1 i zdjąć szczyt. Użyjemy fsubrp (reverse subtract).
         fsubrp st1, st0  ; Wykonuje st1 = st0 - st1, po czym ściąga szczyt.
                          ; Teraz nowy st0 to 5*cos(x) - x.

         ; --- WYPISYWANIE DOUBLE ---
         sub esp, 8       ; Przygotowanie miejsca na printf
         fstp qword [esp] 

         call getaddr
format:
         db "f(x) = %f", 0xA, 0
getaddr:

         call [ebp+3*4]   ; printf
         add esp, 12      ; 8 bajtów (double) + 4 bajty (wsk. na format)

         push 0
         call [ebp+0*4]   ; exit(0)
