        [bits 32]

a1       equ 5           ; Pierwszy wyraz ciągu (np. 5)

Zadanie 4 – r_wyraz.asm

Napisz program r_wyraz.asm obliczający rekurencyjnie n-ty wyraz ciągu arytmetycznego dla zadanych stałych całkowitych a1, r, n. wyraz(1) = a(1) wyraz(n) = wyraz(n-1) + r


        [bits 32]

a1       equ 5           ; Pierwszy wyraz
r        equ 3           ; Różnica
n        equ 4           ; Szukany n-ty wyraz

         jmp start       ; Omijamy kod funkcji

; ==========================================
; FUNKCJA REKURENCYJNA
; Wejście: ECX = n
; Wyjście: EAX = wynik
; ==========================================
oblicz_rek:
         cmp ecx, 1      ; Sprawdzenie warunku stopu (n <= 1)
         jle bazowy
         
         push ecx        ; Zapisanie obecnego 'n' na stos
         dec ecx         ; Zmniejszenie 'n' (n-1)
         call oblicz_rek ; Wywołanie rekurencyjne dla n-1
         pop ecx         ; Odtworzenie 'n'
         
         ; Po powrocie z rekurencji w EAX mamy wartość wyraz(n-1)
         add eax, r      ; eax = wyraz(n-1) + r
         ret

bazowy:
         mov eax, a1     ; Dla n=1 zwracamy po prostu a1
         ret

; ==========================================
; GŁÓWNY PROGRAM
; ==========================================
start:
         mov ebp, ebx    ; Wskaźnik API
         
         mov ecx, n      ; Przekazanie 'n' do funkcji
         call oblicz_rek ; Start machiny
         
         push eax        ; Wynik -> stos
         
         call getaddr
format:
         db "Wyraz (rekurencja) = %d", 0xA, 0
getaddr:

         call [ebp+3*4]  ; printf
         add esp, 4
         
         push 0
         call [ebp+0*4]  ; exit(0)

Zadanie 5 – funkcja.asm

Napisz program funkcja.asm obliczający wartość funkcji f(x) = x - 3*sin(x) dla argumentu zmienno-pozycyjnego x typu double.


        [bits 32]

; f(x) = x - 3*sin(x)
x        equ __?float64?__(1.5)  ; Przykładowy argument x

;        1. Umiesc dane w pamieci procesu i je pomin
         call getaddr

addr_x   dq x       ; Zmienna x
addr_3   dq 3.0     ; Stala 3.0 (leży w pamięci pod adresem: addr_x + 8)

getaddr:
;        esp -> [addr_x][ret]

;        2. Zaladuj adres danych do EBP
         mov ebp, [esp]  ; ebp = wskaźnik na nasze zmienne

         finit           ; fpu init

;        3. Wykonaj obliczenia przy pomocy FPU
         fld qword [ebp]    ; st0 = x
         fld st0            ; Kopiujemy szczyt: st0 = x, st1 = x
         
         fsin               ; Liczymy sinus: st0 = sin(x), st1 = x
         
         ; Mnożymy od razu przez 3.0 leżące w pamięci pod [ebp+8]
         fmul qword [ebp+8] ; st0 = 3.0 * sin(x), st1 = x
         
         ; Mamy teraz na stosie: st0 = 3*sin(x) oraz st1 = x.
         ; Wzór to (x - 3*sin(x)), czyli od st1 odejmujemy st0.
         ; Używamy fsubp, które odejmuje st0 od st1, zapisuje wynik w st1 i zdejmuje szczyt stosu.
         fsubp st1, st0     ; Po tej operacji nowy st0 = x - 3*sin(x)

;        4. Zapisz dane ze stosu FPU do pamieci
         sub esp, 4         ; Robimy miejsce (4 bajty z call + te 4 = 8 bajtów na double)
         
;        esp -> [ ][ ][ret]
         fstp qword [esp]   ; Zapisujemy calkowity wynik na stos

;        5. Wypisywanie
         call getaddr2
format:
         db "f(x) = %f", 0xA, 0
getaddr2:

;        esp -> [format][fl][fh][ret]
         call [ebx+3*4]  ; printf(format, f)
         add esp, 3*4    ; Czyszczenie stosu ze wskaźnika i liczby double (12 bajtów)

         push 0          
         call [ebx+0*4]  ; exit(0)

        [bits 32]

; f(x) = 5*cos(x) - x
x        equ __?float64?__(0.0)  ; Testujemy dla x=0 (wynik powinien byc 5.0)

;        1. Umiesc dane w pamieci procesu i je pomin
         call getaddr

addr_x   dq x       ; Zmienna x
addr_5   dq 5.0     ; Stala 5.0 (adres = addr_x + 8 bajtow)

getaddr:
;        esp -> [addr_x][ret]

;        2. Zaladuj adres do EBP
         mov ebp, [esp]  ; ebp = adres naszych stalych

         finit           ; fpu init

;        3. Obliczenia na FPU
         fld qword [ebp]    ; st0 = x
         fld st0            ; Kopiujemy: st0 = x, st1 = x

         fsubrp st1, st0    ; st0 = 5*cos(x) - x
